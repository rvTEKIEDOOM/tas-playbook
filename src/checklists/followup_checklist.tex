\subsection{1. Evidence \& Data Integrity Review}

\begin{itemize}[leftmargin=*]

    \item $\square$ All onsite findings have supporting evidence\\
    \textit{Mapped SOP: Step 4 -- Onsite Validation and Discovery (Best Practices); Step 5 -- Reporting and Recommendations.}\\
    \textit{myITprocess: Alignment findings, evidence references.}
    \begin{itemize}
        \item Screenshots, photos, logs, or configuration exports captured.
        \item Evidence clearly supports the specific finding (no generic screenshots).
    \end{itemize}

    \item $\square$ Evidence is stored in the correct system\\
    \textit{Mapped SOP: Step 4 -- Onsite Validation and Discovery; Step 6 -- Follow-Up and Continuous Improvement.}\\
    \textit{myITprocess / Tools: ITGlue, myITprocess.}
    \begin{itemize}
        \item ITGlue used for documentation artifacts and diagrams.
        \item myITprocess used for alignment findings and risk justification.
    \end{itemize}

    \item $\square$ Evidence quality supports vCIO decision-making\\
    \textit{Mapped SOP: Step 5 -- Reporting and Recommendations.}\\
    \textit{myITprocess: Notes and supporting evidence.}
    \begin{itemize}
        \item Screenshots show system name, hostname, or timestamps.
        \item Photos are legible and properly oriented.
        \item Logs or reports include sufficient context.
    \end{itemize}

    \item $\square$ No findings rely solely on verbal confirmation\\
    \textit{Mapped SOP: Step 4 -- Best Practices; Step 5 -- Reporting.}
    \begin{itemize}
        \item Verbal statements are clearly labeled as such.
        \item Assumptions are avoided or explicitly noted.
    \end{itemize}

\end{itemize}

\subsection{2. myITprocess Alignment Review}

\begin{itemize}[leftmargin=*]

    \item $\square$ All applicable standards are answered (Yes / No / Unknown)\\
    \textit{Mapped SOP: Step 4 -- Onsite Validation and Discovery; Step 5 -- Reporting and Recommendations.}\\
    \textit{myITprocess: Standards Library, Alignment Review.}
    \begin{itemize}
        \item No standards left blank.
        \item ``Unknown'' used only when evidence could not be obtained.
    \end{itemize}

    \item $\square$ Alignment answers remain factual and non-prescriptive\\
    \textit{Mapped SOP: Step 5 -- Reporting and Recommendations.}
    \begin{itemize}
        \item No remediation or solution language included.
        \item No pricing, scope, or timeline assumptions.
    \end{itemize}

    \item $\square$ Notes clearly explain alignment decisions\\
    \textit{Mapped SOP: Step 5 -- Reporting and Recommendations.}\\
    \textit{myITprocess: Notes / Observations.}
    \begin{itemize}
        \item What was observed.
        \item Why it is aligned or not aligned.
        \item What evidence supports the decision.
    \end{itemize}

    \item $\square$ High-risk or business-impacting items are flagged\\
    \textit{Mapped SOP: Step 5 -- Reporting and Recommendations.}
    \begin{itemize}
        \item Security, backup, DR, or compliance risks identified.
        \item Items needing vCIO prioritization are clearly marked.
    \end{itemize}

\end{itemize}

\subsection{3. Documentation Completion (ITGlue)}

\begin{itemize}[leftmargin=*]

    \item $\square$ ITGlue updated with all verified corrections\\
    \textit{Mapped SOP: Step 4 -- Onsite Validation and Discovery (Update ITGlue); Step 4 -- Step Output.}\\
    \textit{myITprocess / Tools: ITGlue.}
    \begin{itemize}
        \item Asset records updated (model, serial, role, location).
        \item Network diagrams adjusted to match reality.
        \item Server roles and dependencies clarified.
    \end{itemize}

    \item $\square$ Documentation gaps discovered onsite are recorded\\
    \textit{Mapped SOP: Step 4 -- Step Output; Step 6 -- Follow-Up and Continuous Improvement.}
    \begin{itemize}
        \item Missing diagrams, credentials, or vendor information noted.
        \item Gaps are explicitly documented, not silently ignored.
    \end{itemize}

    \item $\square$ No undocumented assumptions remain\\
    \textit{Mapped SOP: Step 6 -- Follow-Up and Continuous Improvement.}
    \begin{itemize}
        \item Anything unverified is clearly labeled as a gap.
    \end{itemize}

\end{itemize}

\subsection{4. Ticket \& Remediation Hygiene}

\begin{itemize}[leftmargin=*]

    \item $\square$ Required remediation tickets are created correctly\\
    \textit{Mapped SOP: Step 4 -- Best Practices (Ticket Creation); Step 5 -- Reporting.}\\
    \textit{myITprocess / Tools: myITprocess, BMS.}
    \begin{itemize}
        \item Tickets created in myITprocess (sync to BMS).
        \item No remediation performed without approval.
    \end{itemize}

    \item $\square$ Tickets contain sufficient context\\
    \textit{Mapped SOP: Step 5 -- Reporting and Recommendations.}
    \begin{itemize}
        \item Clear issue description.
        \item Impacted systems and business context.
        \item Reference to alignment finding or evidence.
    \end{itemize}

    \item $\square$ Ticket priority reflects actual risk\\
    \textit{Mapped SOP: Step 5 -- Reporting and Recommendations.}
    \begin{itemize}
        \item Urgent security or availability risks escalated.
        \item Lifecycle or optimization items left as alignment findings.
    \end{itemize}

\end{itemize}

\subsection{5. Client Interaction \& Expectation Check}

\begin{itemize}[leftmargin=*]

    \item $\square$ No final recommendations were presented to the client\\
    \textit{Mapped SOP: Step 4 -- Best Practices; Step 5 -- Reporting (vCIO approval).}
    \begin{itemize}
        \item No commitments to pricing, timelines, or scope.
    \end{itemize}

    \item $\square$ Client statements about impact or tolerance documented verbatim\\
    \textit{Mapped SOP: Step 4 -- Best Practices; Step 5 -- Reporting.}
    \begin{itemize}
        \item Downtime tolerance.
        \item Data loss tolerance.
        \item Operational pain points.
    \end{itemize}

    \item $\square$ Client concerns captured neutrally\\
    \textit{Mapped SOP: Step 5 -- Reporting and Recommendations.}
    \begin{itemize}
        \item Documented without interpretation or argument.
        \item Preserved for vCIO context.
    \end{itemize}

\end{itemize}

    \item $\square$ Alignment findings are logically grouped\\
    \textit{Mapped SOP: Step 5 -- Reporting and Recommendations.}\\

    \begin{itemize}
        \item Core Infrastructure.
        \item Server Infrastructure.
        \item Security \& Privacy.

        \item Documentation.

    \item $\square$ Key risk themes are immediately identifiable\\
    \textit{Mapped SOP: Step 5 -- Reporting and Recommendations.}
    \begin{itemize}
        \item Aging infrastructure.
        \item Single points of failure.
        \item Security posture weaknesses.
        \item Governance gaps.
    \end{itemize}

    \item $\square$ vCIO has everything needed to finalize strategy\\
    \textit{Mapped SOP: Step 5 -- Reporting and Recommendations.}

        \item No unanswered critical questions.
        \item No undocumented assumptions.
    \end{itemize}

\end{itemize}
